%%file euler_na.m
function [tlist,w] = euler_na(f,x0,dt,ti,tf)
    nn = floor((tf - ti)/dt);
    dtfinal = tf - (ti + nn*dt);
    if dtfinal==0
        w = zeros(nn+1,1);
        tlist = ti:dt:tf;
    else
        w = zeros(nn+2,1);
        tlist = [ti:dt:tf, tf];
    end
    w(1) = x0;
    for i = 1:nn
        w(i+1) = w(i) + dt*f(tlist(i), w(i));
%         fprintf('Step %d:  w = %18.15f\n', i, w(i+1));
    end
    if dtfinal ~= 0
        w(nn+2) = w(nn+1) + dt*f(tlist(nn+1),w(nn+1));
%         fprintf('Step %d:  w = %18.15f\n', nn+2, w(nn+2));
    end
end

%%file rk4.m
function [tlist,w] = rk4(f,x0,dt,ti,tf)
    nn = floor((tf - ti)/dt);
    dtfinal = tf - (ti + nn*dt);
    if dtfinal==0
        w = zeros(nn+1,1);
        tlist = ti:dt:tf;
    else
        w = zeros(nn+2,1);
        tlist = [ti:dt:tf, tf];
    end
    w(1) = x0;
    for i = 1:nn
        k1 = dt*f(w(i));
        k2 = dt*f(w(i) + 0.5.*k1);
        k3 = dt*f(w(i) + 0.5.*k2);
        k4 = dt*f(w(i) + k3);
        w(i+1) = w(i) + (k1 + 2*k2 + 2*k3 + k4)/6.;
        %fprintf('Step %d:  w = %18.15f\n', i, w(i+1));
    end
    if dtfinal ~= 0
        dt = dtfinal;
        k1 = dt*f(w(nn+1));
        k2 = dt*f(w(nn+1) + 0.5.*k1);
        k3 = dt*f(w(nn+1) + 0.5.*k2);
        k4 = dt*f(w(nn+1) + k3);
        w(nn+2) = w(nn+1) + (1/6)*(k1 + 2*k2 + 2*k3 + k4);
        %fprintf('Step %d:  w = %18.15f\n', nn+2, w(nn+2));
    end
end

function [tlist,w] = rk4_na(f,x0,dt,ti,tf)
    nn = floor((tf - ti)/dt);
    dtfinal = tf - (ti + nn*dt);
    if dtfinal==0
        w = zeros(nn+1,1);
        tlist = ti:dt:tf;
    else
        w = zeros(nn+2,1);
        tlist = [ti:dt:tf, tf];
    end
    w(1) = x0;
    for i = 1:nn
        k1 = dt*f(tlist(i),w(i));
        k2 = dt*f(tlist(i)+0.5*dt,w(i) + 0.5.*k1);
        k3 = dt*f(tlist(i)+0.5*dt,w(i) + 0.5.*k2);
        k4 = dt*f(tlist(i)+dt,w(i) + k3);
        w(i+1) = w(i) + (k1 + 2*k2 + 2*k3 + k4)/6.;
%         fprintf('Step %d:  w = %18.15f\n', i, w(i+1));
    end
    if dtfinal ~= 0
        k1 = dt*f(tlist(i),w(i));
        k2 = dt*f(tlist(i)+0.5*dt,w(i) + 0.5.*k1);
        k3 = dt*f(tlist(i)+0.5*dt,w(i) + 0.5.*k2);
        k4 = dt*f(tlist(i)+dt,w(i) + k3);

%         fprintf('Step %d:  w = %18.15f\n', nn+2, w(nn+2));

end

Multi-step Methods¶

Two-step (explicit) Adams-Bashforth Method¶

We need two starting points

$$ w_0 = x_0,\quad w_1= x_1, $$

where $x_1$ is computed first from $x_0$ using a simple time-stepping method. Then the explicit two-step Adams-Bashforth Method is given by

$$ w_{i+1} = w_i + \frac{1}{2}\Delta t \left(3f(t_i,w_i) - f(t_{i-1},w_{i-1}) \right). $$

%%file ab2_na.m
function [tlist,w] = ab2_na(f,x0,dt,ti,tf)
    nn = floor((tf - ti)/dt);
    dtfinal = tf - (ti + nn*dt);
    if dtfinal==0
        w = zeros(nn+1,1);
        tlist = ti:dt:tf;
    else
        w = zeros(nn+2,1);
        tlist = [ti:dt:tf, tf];
    end
    w(1) = x0;
    % Do simple Euler to compute the first-iterate:
    w(2) = w(1) + dt*f(tlist(1),w(1));
    for i = 2:nn
        w(i+1) = w(i) + 0.5*dt*(3*f(tlist(i), w(i))-f(tlist(i-1), w(i-1)));
%         fprintf('Step %d:  w = %18.15f\n', i, w(i+1));
    end
    if dtfinal ~= 0
        w(nn+2) = w(nn+1) + 0.5*dt*( 3*f(tlist(nn+1), w(nn+1)) - f(tlist(nn), w(nn+1)) );
%         fprintf('Step %d:  w = %18.15f\n', nn+2, w(nn+2));
    end
end

Four-step (explicit) Adams-Bashforth Method¶

We need four starting points

$$ w_0 = x_0,\quad w_1= x_1, \quad w_2= x_3, \quad w_3= x_3, $$

where $x_k$, $k=1,2,3$ are computed first from $x_0$ using another time-stepping method. Then the explicit four-step Adams-Bashforth Method is given by

$$ w_{i+1} = w_i + \frac{1}{24}\Delta t \left( 55 f(t_i,w_i) - 59 f(t_{i-1},w_{i-1} + 37 f(t_{i-2},w_{i-2}) - 9 f(t_{i-3},w_{i-3} \right), $$

for $i=3,\dots$.

function [tlist,w] = ab4_na(f,x0,dt,ti,tf)
    nn = floor((tf - ti)/dt);
    dtfinal = tf - (ti + nn*dt);
    if dtfinal==0
        w = zeros(nn+1,1);
        tlist = ti:dt:tf;
    else
        w = zeros(nn+2,1);
        tlist = [ti:dt:tf, tf];
    end
    w(1) = x0;
    % Do RK4 to compute the first 3 iterates:
    tftemp=(ti+3*dt)
    tstart = ti:dt:tftemp;
    wstart = zeros(3,1);    
    [tstart, wstart] = rk4_na(f,x0,dt,ti,tftemp);
    w(1:4)=wstart(1:4);
    for i = 4:nn
        w(i+1) = w(i) + (1/24)*dt*(55*f(tlist(i), w(i)) - 59*f(tlist(i-1), w(i-1)) + 37*f(tlist(i-2), w(i-2)) - 9*f(tlist(i-3), w(i-3)));
%         fprintf('Step %d:  w = %18.15f\n', i, w(i+1));

%         fprintf('Step %d:  w = %18.15f\n', nn+2, w(nn+2));

Can also solve the IVP for a linear second order ODE:

$$ y''(t) + y'(t) - 6 y(t) =0,\quad y(0)=3,\quad y'(0)=1. $$

Set $z(t):=y'(t)$. Then $z'(t)=-z(t)+6y(t)$.

$$ \mathbf{x}(t):=\begin{pmatrix} y(t)\\z(t) \end{pmatrix}. $$

Then the IVP problem is equivalent to the following (autonomous) first order system:

$$ \mathbf{x}'(t) = \begin{pmatrix} 0 & 1 \\ 6 & -1 \end{pmatrix}\mathbf{x}(t) $$

Then the exact solution is

$$ \mathbf{x}(t) = \begin{pmatrix} e^{-3t}+2e^{2t}\\-3e^{-3t}+4e^{2t}\end{pmatrix}. $$

So that $y(t)=e^{-3t}+2e^{2t}$.

%%file rk4_ND.m
function [tlist,w] = rk4(f,x0,dt,ti,tf,Ndim)
    nn = floor((tf - ti)/dt);
    dtfinal = tf - (ti + nn*dt);
    if dtfinal==0
        w = zeros(nn+1,Ndim);
        tlist = ti:dt:tf;
    else
        w = zeros(nn+2,Ndim);
        tlist = [ti:dt:tf, tf];
    end
    w(1,:) = x0;
    for i = 1:nn
        k1 = dt*f(w(i,:));
        k2 = dt*f(w(i,:) + 0.5.*k1);
        k3 = dt*f(w(i,:) + 0.5.*k2);
        k4 = dt*f(w(i,:) + k3);
        w(i+1,:) = w(i,:) + (k1 + 2*k2 + 2*k3 + k4)/6.;
        %fprintf('Step %d:  w = %18.15f\n', i, w(i+1));
    end
    if dtfinal ~= 0
        dt = dtfinal;
        k1 = dt*f(w(nn+1,:));
        k2 = dt*f(w(nn+1,:) + 0.5.*k1);
        k3 = dt*f(w(nn+1,:) + 0.5.*k2);
        k4 = dt*f(w(nn+1,:) + k3);
        w(nn+2,:) = w(nn+1,:) + (1/6)*(k1 + 2*k2 + 2*k3 + k4);
        %fprintf('Step %d:  w = %18.15f\n', nn+2, w(nn+2));
    end
end
